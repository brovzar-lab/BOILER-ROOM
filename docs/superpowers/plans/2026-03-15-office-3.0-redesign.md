**Tech Stack:** TypeScript, React, Zustand, HTML5 Canvas, LimeZu 16×16 sprite sheets, Vitest

```typescript
// src/engine/__tests__/furniture.test.ts
import { FURNITURE, DECORATIONS, ROOM_RUGS } from '../furniture';
import { ROOMS } from '../officeLayout';
import { LIMEZU_ATLAS } from '../limeZuAtlas';

describe('furniture data', () => {
  it('every furniture item references a valid atlasKey', () => {
    for (const item of FURNITURE) {
      expect(LIMEZU_ATLAS[item.atlasKey]).toBeDefined();
    }
  });

  it('every furniture item references a valid roomId', () => {
    const roomIds = ROOMS.map(r => r.id);
    for (const item of FURNITURE) {
      expect(roomIds).toContain(item.roomId);
    }
  });

  it('every decoration references a valid atlasKey', () => {
    for (const dec of DECORATIONS) {
      expect(LIMEZU_ATLAS[dec.atlasKey]).toBeDefined();
    }
  });

  it('furniture items are within their room bounds', () => {
    for (const item of FURNITURE) {
      const room = ROOMS.find(r => r.id === item.roomId);
      if (!room) continue;
      const r = room.tileRect;
      expect(item.col).toBeGreaterThanOrEqual(r.col);
      expect(item.row).toBeGreaterThanOrEqual(r.row);
      expect(item.col + item.width).toBeLessThanOrEqual(r.col + r.width);
      expect(item.row + item.height).toBeLessThanOrEqual(r.row + r.height);
    }
  });

  it('has furniture in every agent office', () => {
    const offices = ['isaac', 'billy', 'patrik', 'marcos', 'sandra', 'charlie', 'wendy'];
    for (const id of offices) {
      const items = FURNITURE.filter(f => f.roomId === id);
      expect(items.length).toBeGreaterThan(0);
    }
  });
});
```

```typescript
// src/engine/furniture.ts
import type { FurnitureItem, DecorationItem, RoomRug } from './types';

// Helper: offset from room origin
function fi(
  id: string, atlasKey: string,
  col: number, row: number,
  width: number, height: number,
  roomId: string
): FurnitureItem {
  return { id, atlasKey, col, row, width, height, roomId };
}

function di(
  id: string, atlasKey: string,
  col: number, row: number,
  width: number, height: number,
  roomId: string
): DecorationItem {
  return { id, atlasKey, col, row, width, height, roomId };
}

// --- ISAAC (top-left: col 1, row 2, 9×9) ---
const ISAAC_FURNITURE: FurnitureItem[] = [
  fi('isaac-desk',    'desk-wood-2wide',  4, 5, 2, 1, 'isaac'),
  fi('isaac-chair',   'chair-office',     5, 6, 1, 1, 'isaac'),
  fi('isaac-shelf',   'bookshelf-2tall',  2, 3, 2, 2, 'isaac'),
  fi('isaac-cabinet', 'filing-cabinet',   8, 3, 1, 2, 'isaac'),
  fi('isaac-plant',   'plant-potted',     2, 9, 1, 1, 'isaac'),
];

export const DECORATIONS: DecorationItem[] = [
```
