3. **src/types/persistence.ts**: Define the PersistenceAdapter interface:
   ```typescript
   export type StoreName = 'conversations' | 'messages' | 'deals' | 'files' | 'memory';

   export interface PersistenceAdapter {
     get<T>(store: StoreName, key: string): Promise<T | undefined>;
     set<T>(store: StoreName, key: string, value: T): Promise<void>;

     getAll<T>(store: StoreName): Promise<T[]>;

   }
   ```

4. **src/services/persistence/adapter.ts**: Re-export the interface and provide a singleton accessor:
   ```typescript
   import type { PersistenceAdapter } from '@/types/persistence';
   import { IndexedDBAdapter } from './indexeddb';

   let instance: PersistenceAdapter | null = null;

   export function getPersistence(): PersistenceAdapter {
     if (!instance) {
       instance = new IndexedDBAdapter();
     }
     return instance;
   }

   export type { PersistenceAdapter };

     conversations: Record<string, Conversation>;
     activeConversationId: string | null;
     streaming: StreamingState;
     error: string | null;

     // Actions
     loadConversations: () => Promise<void>;
     getOrCreateConversation: (agentId: AgentId) => Promise<string>;
     addMessage: (conversationId: string, message: Omit<Message, 'id' | 'timestamp'>) => Promise<void>;
     updateStreamingContent: (content: string) => void;
     startStreaming: (abortController: AbortController) => void;
     stopStreaming: () => void;
     setError: (error: string | null) => void;
     updateConversationTokens: (conversationId: string, tokens: number) => void;
     setConversationSummary: (conversationId: string, summary: string, summaryTokens: number) => void;
   }
   ```
   - Use `create` from Zustand (not `createStore` -- use the hook pattern)
   - `loadConversations`: reads all from IndexedDB via `getPersistence().getAll('conversations')`, populates state
   - `addMessage`: generates UUID via `crypto.randomUUID()`, persists message to IndexedDB, updates conversation in state and IndexedDB
   - `getOrCreateConversation`: if conversation exists for agentId, return its id. Otherwise create new Conversation, persist to IndexedDB, return id
   - Streaming actions: manage `StreamingState` (isStreaming, currentContent, abortController)
   - All mutations that persist data call `getPersistence()` methods
