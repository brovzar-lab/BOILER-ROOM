```typescript
// Layer 2b: Area rugs (after floor tiles + tint, before walls)
for (const rug of ROOM_RUGS) {
  const x = rug.col * TILE_SIZE;
  const y = rug.row * TILE_SIZE;
  const w = rug.w * TILE_SIZE;
  const h = rug.h * TILE_SIZE;
  // Fill
  ctx.fillStyle = rug.color;
  ctx.fillRect(x, y, w, h);
  // 1px woven edge border
  ctx.strokeStyle = rug.borderColor;
  ctx.lineWidth = 1;
  ctx.strokeRect(x + 0.5, y + 0.5, w - 1, h - 1);
  // Inner border pattern (woven suggestion): second border 2px inset
  ctx.strokeStyle = rug.borderColor;
  ctx.setLineDash([2, 2]);
  ctx.strokeRect(x + 2.5, y + 2.5, w - 5, h - 5);
  ctx.setLineDash([]);
}
```
