**renderer.ts — Draw speech bubble for needs-attention agents:**

Add a new render layer between characters and UI overlays (or just after characters). After the character sort-and-render loop, iterate characters again for overlays:

For each character where `agentStatuses[ch.id] === 'needs-attention'`:
- Draw a small speech bubble icon above the character's head
- Position: above the character rectangle, centered horizontally
- Design: Small white rounded rectangle (6x5 pixels at zoom=1) with a tiny triangle pointer at bottom center, filled white
- At zoom=1: bubble is small but visible. At zoom=2: bubble is clear.

Implementation:
```typescript
function renderStatusOverlays(
  ctx: CanvasRenderingContext2D,
  characters: Character[],
  agentStatuses: Record<string, string>,
  tileSize: number,
  offsetX: number,
  offsetY: number,
  zoom: number,
): void {
  for (const ch of characters) {
    if (ch.id === 'billy') continue;
    const status = agentStatuses[ch.id];

    if (status === 'needs-attention') {
      // Speech bubble above character
      const cx = Math.floor(ch.x * zoom + offsetX + (TILE_SIZE * zoom) / 2);
      const bubbleY = Math.floor(ch.y * zoom + offsetY - 4 * zoom);
      const bw = Math.floor(6 * zoom);
      const bh = Math.floor(5 * zoom);
      const radius = Math.max(1, Math.floor(zoom));

      // Bubble body
      ctx.fillStyle = '#ffffff';
      ctx.beginPath();
      ctx.roundRect(cx - bw / 2, bubbleY - bh, bw, bh, radius);
      ctx.fill();

      // Small triangle pointer
      const triSize = Math.max(1, Math.floor(zoom));
      ctx.beginPath();
      ctx.moveTo(cx - triSize, bubbleY);
      ctx.lineTo(cx + triSize, bubbleY);
      ctx.lineTo(cx, bubbleY + triSize);
      ctx.closePath();
      ctx.fill();

      // Small dot inside bubble (notification indicator)
      ctx.fillStyle = '#f87171'; // Red dot
      const dotR = Math.max(1, Math.floor(zoom * 0.8));
      ctx.beginPath();
      ctx.arc(cx, bubbleY - bh / 2, dotR, 0, Math.PI * 2);
      ctx.fill();
    }

    // For 'thinking' status: the work animation already handles this visually
    // Optionally add a subtle "..." dots above thinking agents
    if (status === 'thinking') {
      const cx = Math.floor(ch.x * zoom + offsetX + (TILE_SIZE * zoom) / 2);
      const dotsY = Math.floor(ch.y * zoom + offsetY - 2 * zoom);
      const dotR = Math.max(1, Math.floor(zoom * 0.6));
      const spacing = Math.floor(2.5 * zoom);

      ctx.fillStyle = '#fbbf24'; // Amber dots for thinking
      for (let i = -1; i <= 1; i++) {
        ctx.beginPath();
        ctx.arc(cx + i * spacing, dotsY, dotR, 0, Math.PI * 2);
        ctx.fill();
      }
    }
  }
}
```
